                //Represents anything that can be iterated over
interface Iterable {
//Returns an iterator over this collection
    Iterator iterator();
}

// represent IListIterator to use iterator for ilist
class IListIterator implements Iterator {
    IList items;
    // constructor
    IListIterator(IList items) {
        this.items = items;
    }
    // Does this list have at least one more value?
    public boolean hasNext() {
        return this.items.isCons();
    }
     // Get the next value in this list
    // EFFECT: Advance the iterator to the subsequent value
    public T next() {
        if (this.hasNext()) {
            Cons itemsasCons = this.items.asCons();
            T answer = itemsasCons.first;
            this.items = itemsasCons.rest;
            return answer;
        }
        else {
            throw new RuntimeException("There is no next item");
        }
    }
    
    // EFFECT: Remove the item just returned by next()
    // NOTE: This method may not be supported by every iterator; ignore it for
    // now
    public void remove() {
        throw new UnsupportedOperationException("Don't do this!");
    }
    
}



//represent a abstract list
interface IList extends Iterable {
    // add a new T to the IList
    IList add(T that);
    
    // check if the ilist is a cons list
    boolean isCons();
    
    // return the cons of the ilist
    Cons asCons();
    
    // get a list's size
    int size();
    
    // remove one from a list
    IList remove(T that);
    
   // if the piece has been picked, remove it from the list
    IList pickedUp(Cell that);
    
    
}

//represent a abstract empty list
class Empty implements IList {

    // constructor
    Empty() {
     // nothing to do!
    }
 
    // iterator in the empty list
    public Iterator iterator() {
        return new IListIterator(this);
    }
 
    // add(T) in the empty list
    public IList add(T that) {
        return new Cons(that, this);
    }
 
    // empty is not a cons, return false
    public boolean isCons() {
        return false;
    }

    // empty is not a cons, return exception
    public Cons asCons() {
        throw new ClassCastException("Empty is not a cons");
    }
    
    //size in the empty list
    public int size() {
        return 0;
    }
    
    // remove in the empty
    public IList remove(T that) {
        return new Empty();
    }
    
    // if the piece has been picked, remove it from the list
    // in the empty list
    public IList pickedUp(Cell player) {
        return new Empty();
    }

}

//represent a abstract cons list
class Cons implements IList {
    T first;
    IList rest;

    // constructor
    Cons(T first, IList rest) {
        this.first = first;
        this.rest = rest;
    }
 
    // the iterator in the cons list
    public Iterator iterator() {
        return new IListIterator(this);
    }
 
    //add(T) in the cons list
    public IList add(T that) {
        return new Cons(that, this);
    }
 
    // check if cons is a cons list, return true
    public boolean isCons() {
        return true;
    }

   // return the cons type of the cons list
    public Cons asCons() {
        return this;
    }
    
    // size in the cons list
    public int size() {
        return 1 + this.rest.size();
    }

    // remove in the cons list
    public IList remove(T that) {
        if (that == this.first) {
            return this.rest;
        } 
        else {
            return new Cons(this.first, this.rest.remove(that));
        }

    }
    
    // if the piece has been picked, remove it from the list
    // in the cons list
    public IList pickedUp(Cell player) {
        if (((Target) this.first).picked(player)) {
            return this.remove(this.first);
        }
        else {
            return new Cons(this.first, this.rest.pickedUp(player));
        }
    }
    
 
}
